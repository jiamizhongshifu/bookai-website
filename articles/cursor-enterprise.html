1.2 安全配置指南

以下是Cursor企业版的推荐安全配置：

// cursor.config.js
module.exports = {
    security: {
        encryption: {
            enabled: true,
            algorithm: 'AES-256-GCM'
        },
        dataProcessing: {
            localOnly: true,
            sensitivePatterns: [
                '密码',
                'API_KEY',
                'TOKEN'
            ]
        },
        networkPolicy: {
            allowedDomains: ['*.company.com'],
            restrictedPaths: ['/sensitive/*']
        }
    }
}
